async function cargarCSV(){ try { const resp = await fetch('tesoreria.csv?t=' + Date.now()); const txt = await resp.text(); // CORRECCIÓN: Usar \n para separar las filas del archivo const filas = txt.trim().split("\n"); const headers = filas[0].split(";"); const datos = filas.slice(1).map(row => { const c = row.split(";"); let o = {}; headers.forEach((h, i) => { if (h) o[h.trim()] = c[i] ? c[i].trim() : ""; }); // Convertimos el saldo a número let saldoOriginal = parseInt(o['Saldo Vencido']) || 0; // Tu lógica actual: si el saldo es negativo o cero, muestra 0 deuda o['Deuda Vencida'] = saldoOriginal > 0 ? saldoOriginal : 0; return o; }); return datos; } catch (e) { console.error("Error cargando el CSV:", e); return []; } }
